روش ۱ — نصب خودکار (پیشنهادی)

با SSH به سرور وصل شوید و این دستورات را یکی‌یکی بزنید:

قدم ۱ — وصل شدن به سرور

ssh root@IP_SERVER_SHOMA

به‌جای IP_SERVER_SHOMA آی‌پی واقعی VPS را بگذارید.

قدم ۲ — دریافت اسکریپت نصب

apt update && apt install -y git
cd /root
git clone https://github.com/aradesign/hr-manager-v2.git
cd hr-manager-v2
chmod +x scripts/vps-install.sh

قدم ۳ — اجرای نصب

sudo bash scripts/vps-install.sh

از شما می‌پرسد:

  • دامنه — مثلاً hr.example.com
  • رمز دیتابیس — یک رمز قوی انتخاب کنید و یادداشت کنید
  • ایمیل SSL — برای گواهی رایگان (Enter = بدون SSL فعلاً)

پایان نصب → بروید به https://دامنه-شما/admin/login

ورود اول: admin@example.com / password — بلافاصله رمز را عوض کنید.

روش ۲ — نصب دستی (قدم‌به‌قدم)

اگر اسکریپت خودکار خطا داد، این مراحل را دستی انجام دهید.

۱) اتصال و به‌روزرسانی

ssh root@IP_SERVER_SHOMA
apt update && apt upgrade -y

۲) Swap (برای سرور ۱GB RAM ضروری است)

fallocate -l 2G /swapfile
chmod 600 /swapfile
mkswap /swapfile
swapon /swapfile
echo '/swapfile none swap sw 0 0' >> /etc/fstab

۳) نصب Nginx، Git، MariaDB، PHP

apt install -y nginx mariadb-server git curl unzip software-properties-common
add-apt-repository ppa:ondrej/php -y
apt update
apt install -y php8.3-fpm php8.3-cli php8.3-mysql php8.3-mbstring php8.3-xml php8.3-curl php8.3-zip php8.3-gd php8.3-bcmath php8.3-intl

۴) نصب Composer

curl -sS https://getcomposer.org/installer | php
mv composer.phar /usr/local/bin/composer
composer --version

۵) ساخت دیتابیس

mysql -u root
CREATE DATABASE leili_hr CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
CREATE USER 'leili'@'localhost' IDENTIFIED BY 'RAMZ_GHALI';
GRANT ALL PRIVILEGES ON leili_hr.* TO 'leili'@'localhost';
FLUSH PRIVILEGES;
EXIT;

۶) دریافت پروژه از GitHub

اشتباه رایج: اگر پوشه خالی نیست، git clone ... . خطا می‌دهد. حتماً پوشه را اول خالی کنید. 
mkdir -p /var/www/leili-hr
cd /var/www/leili-hr
# اگر قبلاً چیزی اینجا بود:
# rm -rf /var/www/leili-hr/*
git clone https://github.com/aradesign/hr-manager-v2.git .

بعد از clone باید فایل artisan را ببینید:

ls -la artisan composer.json

۷) فایل .env

cp .env.example .env
nano .env

این خطوط را حتماً تنظیم کنید:

APP_ENV=production
APP_DEBUG=false
APP_URL=https://hr.example.com

DB_CONNECTION=mysql
DB_HOST=127.0.0.1
DB_PORT=3306
DB_DATABASE=leili_hr
DB_USERNAME=leili
DB_PASSWORD=RAMZ_GHALI

SESSION_DRIVER=database
CACHE_STORE=database
QUEUE_CONNECTION=database
php artisan key:generate

۸) نصب وابستگی‌ها و دیتابیس

composer install --no-dev --optimize-autoloader
php artisan migrate --force
php artisan db:seed --force
php artisan storage:link
php artisan config:cache
php artisan route:cache
php artisan view:cache

۹) دسترسی پوشه‌ها

chown -R www-data:www-data storage bootstrap/cache
chmod -R ug+rwx storage bootstrap/cache

روش ۳ — اگر git clone نشد (ZIP)

گاهی git روی سرور مشکل دارد. به‌جای clone:

cd /var/www
apt install -y unzip wget
wget -O hr.zip https://github.com/aradesign/hr-manager-v2/archive/refs/heads/main.zip
unzip hr.zip
mv hr-manager-v2-main leili-hr
cd leili-hr

بعد از این، از قدم ۷ روش دستی ادامه دهید (.env و composer و ...).

تنظیم Nginx

nano /etc/nginx/sites-available/leili-hr
server {
    listen 80;
    server_name hr.example.com;
    root /var/www/leili-hr/public;

    index index.php;
    charset utf-8;

    location / {
        try_files $uri $uri/ /index.php?$query_string;
    }

    location ~ \.php$ {
        fastcgi_pass unix:/var/run/php/php8.3-fpm.sock;
        fastcgi_param SCRIPT_FILENAME $realpath_root$fastcgi_script_name;
        include fastcgi_params;
    }

    location ~ /\.(?!well-known).* {
        deny all;
    }
}
ln -s /etc/nginx/sites-available/leili-hr /etc/nginx/sites-enabled/
rm -f /etc/nginx/sites-enabled/default
nginx -t
systemctl reload nginx

SSL

apt install -y certbot python3-certbot-nginx
certbot --nginx -d hr.example.com

بعد از نصب

  • ورود ادمین: /admin/login
  • import پرسنل و درخواست‌ها از پنل ادمین
  • Cron (اختیاری): * * * * * cd /var/www/leili-hr && php artisan schedule:run
